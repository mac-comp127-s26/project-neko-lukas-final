package breakout;

import edu.macalester.graphics.CanvasWindow;
import edu.macalester.graphics.GraphicsGroup;
import edu.macalester.graphics.GraphicsObject;
import edu.macalester.graphics.Point;
import edu.macalester.graphics.Rectangle;
import java.awt.Color;
import java.awt.event.KeyEvent;
import java.awt.event.MouseEvent;
import java.util.HashMap;
import java.util.Map;

/**
 * The game of Breakout.
 */
public class AABreakoutGame {
    private static final int CANVAS_WIDTH = 600;
    private static final int CANVAS_HEIGHT = 600;
    private static int wait;
    private static CanvasWindow canvas;
    private static GraphicsGroup barrierLayer;
    private static GraphicsGroup brickLayer;
    private static GraphicsGroup ballLayer;

    private static Ball ball;
    private static Point bPOS;
    private static double speed;
    private static String moving;
    private static int startingLives;
    private static int lives;
    private static double brickCount;
    
    private static Barrier westWall;
    private static Barrier eastWall;
    private static Barrier northWall;
    private static Barrier southWall;



    public AABreakoutGame() {
        canvas = new CanvasWindow("Breakout!", CANVAS_WIDTH, CANVAS_HEIGHT);
        canvas.setBackground(new Color(0,0,0));

        //...Set object variables Variables
            barrierLayer = new GraphicsGroup();
            brickLayer = new GraphicsGroup();
            ballLayer = new GraphicsGroup();
            speed = 5;
            moving = "ball OOB";
            startingLives = 3;
            lives = startingLives;
        //...

        //...Construct game objects
            constructBricks(6, 3);
            constructBounds();
            bPOS = new Point(southWall.getGraphics().getCenter().getX() - 50, 
                             southWall.getGraphics().getCenter().getY() - 50);
            constructBall();
        //...

        //...Set platform movement event
            canvas.onMouseMove(event -> {
                double mouseX = event.getPosition().getX();
                double southWallY = CANVAS_HEIGHT - 10 * 3;
                southWall.getGraphics().setCenter(mouseX, southWallY);
            });
        //...
    }

    public static void main(String[] args){
        new AABreakoutGame();
        int n = 0;
            if(n == 0){System.out.println("------------------");}
            if(n < 50){System.out.println("game tick");}
            n += 1;
            System.out.println(lives);
        
        //...Animation
            canvas.animate(() -> {
                //moving variable determines gamestate

                //...Ball is in motion
                    if(moving == "true"){
                        ball.move();
                        barrierCollisionCheck();
                        brickCollisionCheck();
                        outOfBoundsCheck();
                        if(lives < 0){moving = "loss";}
                        if(brickCount <= 0){moving = "win";}
                    }
                //...
                
                //...ball has gone out of bounds
                    else if(moving == "ball OOB"){
                        resetBall();
                        wait++; 
                        if(wait>=180){ //at 60 fps, 3 seconds pass before ball moves again
                            moving = "true";
                            wait = 0;
                            System.out.println("resetBallWAIT");
                        }
                    }
                //...

                //...When all lives are lost
                    else if(moving == "loss"){
                        //...Clear and rest bricks
                            brickLayer.removeAll();
                            constructBricks(1, 3);
                        //...
                        
                        //...Reset game state
                            lives = startingLives;
                            moving = "ball OOB";
                        //...
                    }
                //...

                //...When all bricks are destroyed
                    else if(moving == "win"){
                        System.out.println("win");

                            //...Clear and rest bricks
                                brickLayer.removeAll();
                                constructBricks(20, 2);
                                constructBricks(20, 7);
                            //...
                            
                            //...Reset game state
                                lives = startingLives;
                                moving = "ball OOB";
                            //...
                        }
                //...

                canvas.draw();
            });
        //...
    }

    private static void constructBounds(){
        //...places edge barriers and movable platform (southWall is the platform)
            double bound = 10;
            westWall = new Barrier(0, 0, bound, CANVAS_HEIGHT);
            eastWall = new Barrier(CANVAS_WIDTH - bound, 0, bound, CANVAS_HEIGHT);
            northWall = new Barrier(0, 0, CANVAS_WIDTH, bound);
            //southWall = new Barrier(0, CANVAS_HEIGHT - bound, CANVAS_WIDTH, bound);
            southWall = new Barrier(300, CANVAS_HEIGHT - bound * 3, bound * 8, bound);
            
            barrierLayer.add(westWall.getGraphics());
            barrierLayer.add(eastWall.getGraphics());
            barrierLayer.add(northWall.getGraphics());
            barrierLayer.add(southWall.getGraphics());
            canvas.add(barrierLayer);
        //...
    }

    private static void constructBricks(double columns, double rows){
        brickCount = columns * rows;
        double space  = 5;
        double brickW = (CANVAS_WIDTH - 20 - (space * (columns + 1)))/columns;
        for(double by = 0; by < rows; by++){
            for(double bx = 0; bx < columns; bx++){
                Brick brick1 = new Brick(brickW * bx + space * (bx + 1) + 10, 100 + by * (10 + space), 
                                        brickW, 10);
                brickLayer.add(brick1.getGraphics());
            }
            canvas.add(brickLayer);
        }
    }

    private static void constructBall(){
        ballLayer.removeAll();
        ball = new Ball(bPOS.getX(), bPOS.getY(), 7.5);
        ballLayer.add(ball.getGraphicsGroup());
        ballLayer.add(ball.getContactsGroup());
        canvas.add(ballLayer);
        ball.setSpeed(speed);
    }

    private static void resetBall() {
        //...Places the ball's position above platform
            moving = "ball OOB";
            bPOS = new Point(southWall.getGraphics().getCenter().getX() - 50, 
                            southWall.getGraphics().getCenter().getY() - 50);
            constructBall();
        //...
    }

    private static void barrierCollisionCheck(){
        //...Barrier collision check & deflection
            boolean countactWest = barrierLayer.getElementAt(ball.westCanvasPos()) instanceof Rectangle;
            boolean countactNorth = barrierLayer.getElementAt(ball.northCanvasPos()) instanceof Rectangle;
            boolean countactEast = barrierLayer.getElementAt(ball.eastCanvasPos()) instanceof Rectangle;
            boolean countactSouth = barrierLayer.getElementAt(ball.southCanvasPos()) instanceof Rectangle;
                
            if(countactWest){ball.deflection(1); System.out.println("CONTACT 1");} 
            if(countactNorth){ball.deflection(2); System.out.println("CONTACT 2");}
            if(countactEast){ball.deflection(3); System.out.println("CONTACT 3");}
            if(countactSouth){ball.deflection(4); System.out.println("CONTACT 4");}
        //...
    }

    private static void brickCollisionCheck(){
        //...Brick collision check, deflection, & brick removal
            GraphicsObject brickWest = brickLayer.getElementAt(ball.westCanvasPos());
            GraphicsObject brickNorth = brickLayer.getElementAt(ball.northCanvasPos());
            GraphicsObject brickEast = brickLayer.getElementAt(ball.eastCanvasPos());
            GraphicsObject brickSouth = brickLayer.getElementAt(ball.southCanvasPos());

            if(brickWest != null){ball.deflection(1); System.out.println("CONTACT 1"); brickLayer.remove(brickWest);}
            if(brickNorth != null){ball.deflection(2); System.out.println("CONTACT 2"); brickLayer.remove(brickNorth);}
            if(brickEast != null){ball.deflection(3); System.out.println("CONTACT 3"); brickLayer.remove(brickEast);}
            if(brickSouth != null){ball.deflection(4); System.out.println("CONTACT 4"); brickLayer.remove(brickSouth);}

            if(brickWest != null || brickNorth != null || brickEast != null || brickSouth != null){brickCount--;}
        //...
    }

    private static void outOfBoundsCheck(){
        //...check if ball is out of bounds
            if(ball.getGraphicsGroup().getCenter().getY() > CANVAS_HEIGHT
            || ball.getGraphicsGroup().getCenter().getY() < 0
            || ball.getGraphicsGroup().getCenter().getX() > CANVAS_WIDTH
            || ball.getGraphicsGroup().getCenter().getX() < 0
                ){moving = "ball OOB"; System.out.println("OUT OF BOUNDS"); resetBall(); lives --; System.out.println(lives);}
        //...
    }
}
